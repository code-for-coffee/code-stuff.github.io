class Dog
def initialize(name, breed, age)
@name = name
@breed = breed
@age = age
end
def age
@age
end
def age=(age)
@age = age
end
def bark
"RUFF RUFF"
end
end
spot = Dog.new("Spot", "Corgi", "3")
puts spot.bark
=> "RUFF RUFF"
spot.age = 4
puts spot.age
=> 4
